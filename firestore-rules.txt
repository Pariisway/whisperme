// Firestore Security Rules for Whisper+me
// Copy these rules to Firebase Console > Firestore > Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Users can read/write their own user document
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Users can read/write their own profile
    match /profiles/{userId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Users can read/write their own stats
    match /userStats/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // All users can read call sessions where they are participant
    match /callSessions/{sessionId} {
      allow read: if request.auth != null && 
        (resource.data.whisperId == request.auth.uid || 
         resource.data.callerId == request.auth.uid);
      allow create: if request.auth != null;
      allow update: if request.auth != null && 
        (resource.data.whisperId == request.auth.uid || 
         resource.data.callerId == request.auth.uid);
    }
    
    // Public profiles can be read by anyone, but only owners can write
    match /publicProfiles/{userId} {
      allow read: if true; // Public read access
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Fallback: deny all other access
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
